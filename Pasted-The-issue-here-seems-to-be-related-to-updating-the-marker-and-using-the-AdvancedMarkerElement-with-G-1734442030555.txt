2. Updating the Marker on Address Change
Ensure the marker and map are updated correctly:

typescript

useEffect(() => {
  const updateMarker = async () => {
    if (!map || !marker || !window.google?.maps) {
      console.log('Map or marker not ready yet');
      return;
    }

    // If no address is provided, reset to the default location
    if (!address?.trim()) {
      const defaultLocation = { lat: -33.918861, lng: 18.423300 }; // Cape Town
      map.setCenter(defaultLocation);
      map.setZoom(13);
      marker.setMap(null); // Hide marker
      return;
    }

    try {
      console.log('Geocoding address:', address);
      const geocoder = new window.google.maps.Geocoder();

      const result = await new Promise((resolve, reject) => {
        geocoder.geocode({ address }, (results, status) => {
          if (status === 'OK' && results?.[0]) {
            resolve(results[0]);
          } else {
            reject(new Error(`Geocoding failed: ${status}`));
          }
        });
      });

      const location = result.geometry.location;
      console.log('Found location:', location.toString());

      // Update the map and marker
      map.setCenter(location);
      map.setZoom(16);

      marker.setPosition(location);
      marker.setMap(map);

      console.log('Marker placed successfully at:', location.toString());
    } catch (error) {
      console.error('Error updating marker:', error);
      marker.setMap(null); // Hide marker on error
    }
  };

  updateMarker();
}, [address, map, marker]);
