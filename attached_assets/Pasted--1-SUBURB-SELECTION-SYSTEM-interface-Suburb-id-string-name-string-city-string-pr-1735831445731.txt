// 1. SUBURB SELECTION SYSTEM

interface Suburb {
  id: string;
  name: string;
  city: string;
  province: string;
  coordinates: {
    lat: number;
    lng: number;
  };
  // Additional metadata
  propertyCount?: number;
  averagePrice?: number;
}

// Combobox component for suburb selection
import { Command } from 'cmdk'

function SuburbSearch() {
  const [inputValue, setInputValue] = useState("")
  const [suburbs, setSuburbs] = useState<Suburb[]>([])
  
  // Debounced search function
  const searchSuburbs = useCallback(
    debounce(async (search: string) => {
      if (search.length < 2) return
      const results = await fetch(`/api/suburbs/search?q=${search}`)
      const data = await results.json()
      setSuburbs(data)
    }, 300),
    []
  )

  return (
    <Command>
      <Command.Input
        value={inputValue}
        onValueChange={(value) => {
          setInputValue(value)
          searchSuburbs(value)
        }}
        placeholder="Search for a suburb..."
        className="w-full"
      />
      <Command.List>
        {suburbs.map((suburb) => (
          <Command.Item
            key={suburb.id}
            value={suburb.id}
            onSelect={(value) => {
              // Handle selection
            }}
          >
            {suburb.name}, {suburb.city}
          </Command.Item>
        ))}
      </Command.List>
    </Command>
  )
}

// 2. DATA POINTS MODAL AND DISPLAY

interface DataPoint {
  id: string;
  type: 'news' | 'development' | 'statistic' | 'market_report';
  source: {
    name: string;
    url: string;
    reliability: number; // 1-5 scale
  };
  title: string;
  content: string;
  date: string;
  sentiment: number; // -1 to 1
  category: string;
  relevanceScore: number; // 0-1
  impact: {
    score: number; // -1 to 1
    reasoning: string;
  };
}

interface DataPointsModalProps {
  isOpen: boolean;
  onClose: () => void;
  dataPoints: DataPoint[];
}

function DataPointsModal({ isOpen, onClose, dataPoints }: DataPointsModalProps) {
  const [filter, setFilter] = useState<string>('all')
  const [sort, setSort] = useState<string>('relevance')
  
  const sortedAndFilteredData = useMemo(() => {
    let filtered = dataPoints;
    
    // Apply filters
    if (filter !== 'all') {
      filtered = filtered.filter(dp => dp.type === filter)
    }
    
    // Apply sorting
    return filtered.sort((a, b) => {
      switch (sort) {
        case 'relevance':
          return b.relevanceScore - a.relevanceScore
        case 'date':
          return new Date(b.date).getTime() - new Date(a.date).getTime()
        case 'impact':
          return b.impact.score - a.impact.score
        default:
          return 0
      }
    })
  }, [dataPoints, filter, sort])

  return (
    <Dialog open={isOpen} onOpenChange={onClose}>
      <DialogContent className="max-w-4xl h-[80vh]">
        <DialogHeader>
          <DialogTitle>Analysis Data Points</DialogTitle>
          <DialogDescription>
            Showing {dataPoints.length} data points used in the analysis
          </DialogDescription>
        </DialogHeader>
        
        <div className="flex gap-4 mb-4">
          <Select value={filter} onValueChange={setFilter}>
            <SelectTrigger>
              <SelectValue placeholder="Filter by type" />
            </SelectTrigger>
            <SelectContent>
              <SelectItem value="all">All Types</SelectItem>
              <SelectItem value="news">News</SelectItem>
              <SelectItem value="development">Development</SelectItem>
              <SelectItem value="statistic">Statistics</SelectItem>
              <SelectItem value="market_report">Market Reports</SelectItem>
            </SelectContent>
          </Select>

          <Select value={sort} onValueChange={setSort}>
            <SelectTrigger>
              <SelectValue placeholder="Sort by" />
            </SelectTrigger>
            <SelectContent>
              <SelectItem value="relevance">Relevance</SelectItem>
              <SelectItem value="date">Date</SelectItem>
              <SelectItem value="impact">Impact</SelectItem>
            </SelectContent>
          </Select>
        </div>

        <div className="overflow-y-auto flex-1">
          {sortedAndFilteredData.map((dataPoint) => (
            <Card key={dataPoint.id} className="mb-4">
              <CardHeader>
                <div className="flex justify-between items-start">
                  <div>
                    <CardTitle>{dataPoint.title}</CardTitle>
                    <CardDescription>
                      {dataPoint.source.name} • {new Date(dataPoint.date).toLocaleDateString()}
                    </CardDescription>
                  </div>
                  <Badge variant={getVariantForType(dataPoint.type)}>
                    {dataPoint.type}
                  </Badge>
                </div>
              </CardHeader>
              <CardContent>
                <p className="text-sm text-muted-foreground mb-4">
                  {dataPoint.content}
                </p>
                <div className="grid grid-cols-2 gap-4 text-sm">
                  <div>
                    <span className="font-medium">Relevance Score:</span>{" "}
                    {(dataPoint.relevanceScore * 100).toFixed(1)}%
                  </div>
                  <div>
                    <span className="font-medium">Impact Score:</span>{" "}
                    <span className={getImpactColor(dataPoint.impact.score)}>
                      {dataPoint.impact.score > 0 ? "+" : ""}
                      {dataPoint.impact.score.toFixed(2)}
                    </span>
                  </div>
                </div>
              </CardContent>
            </Card>
          ))}
        </div>
      </DialogContent>
    </Dialog>
  )
}
