// Function to calculate IRR
function calculateIRR(cashFlows) {
    const maxIterations = 1000;
    const tolerance = 0.00001;
    let guess = 0.1;

    function npv(rate) {
        return cashFlows.reduce((sum, cashFlow, index) => 
            sum + cashFlow / Math.pow(1 + rate, index), 0);
    }

    for (let i = 0; i < maxIterations; i++) {
        const npvValue = npv(guess);
        if (Math.abs(npvValue) < tolerance) {
            return guess;
        }

        const derivativeNPV = cashFlows.reduce((sum, cashFlow, index) => 
            sum - index * cashFlow / Math.pow(1 + guess, index + 1), 0);

        const newGuess = guess - npvValue / derivativeNPV;
        if (Math.abs(newGuess - guess) < tolerance) {
            return newGuess;
        }

        guess = newGuess;
    }

    return null; // IRR not found
}

            }, 2000);

            }, 2000);

        }

        // Handle general errors
        const errorMessage = error.message || error.error || 'Unknown error occurred';
        showAlert('Failed to save property: ' + errorMessage, 'danger');
    }

function getCostTables(purchasePrice) {
    // Define maximum values from the cost tables for bond and transfer costs
    const maxBondCost = bondCostsTable[bondCostsTable.length - 1];
    const maxTransferCost = transferCostsTable[transferCostsTable.length - 1];

    if (purchasePrice <= 30000000) {
        // Use existing cost tables for values <= 30 million
        const transferCost = transferCostsTable.find(range => purchasePrice <= range.value);
        const bondCost = bondCostsTable.find(range => purchasePrice <= range.value);

        if (!transferCost) {
            console.warn(`No matching transfer cost found for purchase price: ${purchasePrice}`);
            return null;
        }

        if (!bondCost) {
            console.warn(`No matching bond cost found for purchase price: ${purchasePrice}`);
            return null;
        }

        return {
            bondFee: bondCost.bondFee,
            bondDisbursements: bondCost.disbursements,
            bondVAT: bondCost.vat,
            bondDeedsFee: bondCost.deedsFee,
            transferFee: transferCost.transferFee,
            transferDisbursements: transferCost.disbursements,
            transferVAT: transferCost.vat,
            transferDeedsFee: transferCost.deedsFee,
            transferDuty: transferCost.transferDuty
        };
    } else {
        // For values over R30 million, apply a percentage-based increase
        const extraAmount = purchasePrice - 30000000;

        // Define percentage increase for each additional R1 million (adjustable)
        const percentageIncrease = 0.05; // 5% increase for every R1 million over R30 million

        // Calculate bond fees with incremental increase
        const bondFee = maxBondCost.bondFee + (extraAmount * percentageIncrease);
        const bondVAT = maxBondCost.vat + (extraAmount * percentageIncrease * 0.15);  // Adjusting VAT proportionately

        // Calculate transfer costs with incremental increase
        const transferFee = maxTransferCost.transferFee + (extraAmount * percentageIncrease);
        const transferVAT = maxTransferCost.vat + (extraAmount * percentageIncrease * 0.15);  // Adjusting VAT proportionately
        const transferDuty = maxTransferCost.transferDuty + (extraAmount * 0.1);  // Example: Increase transfer duty by 10% of extra

        return {
            bondFee: bondFee,
            bondDisbursements: maxBondCost.disbursements,
            bondVAT: bondVAT,
            bondDeedsFee: maxBondCost.deedsFee,
            transferFee: transferFee,
            transferDisbursements: maxTransferCost.disbursements,
            transferVAT: transferVAT,
            transferDeedsFee: maxTransferCost.deedsFee,
            transferDuty: transferDuty
        };
    }
}

    function populateAirbnbPerformance(data) {
        const months = ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun', 'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec'];

        let totalRevenueMedium = 0;

            const revenueMedium = monthlyRate * occupancyMedium * daysInMonth;

            document.getElementById(`revenueMedium${month}`).textContent = 'R' + formatNumber(revenueMedium, 0);
            document.getElementById(`revenueHigh${month}`).textContent = 'R' + formatNumber(revenueHigh, 0);

            totalRevenueLow += revenueLow;
            totalRevenueMedium += revenueMedium;
            totalRevenueHigh += revenueHigh;
        });

        const avgOccupancyLow = occupancyLowValues.reduce((sum, value) => sum + value, 0) / occupancyLowValues.length;
        const avgOccupancyMedium = occupancyMediumValues.reduce((sum, value) => sum + value, 0) / occupancyMediumValues.length;
        const avgOccupancyHigh = occupancyHighValues.reduce((sum, value) => sum + value, 0) / occupancyHighValues.length;

        document.getElementById('occupancyMediumAvg').textContent = formatNumber(avgOccupancyMedium, 0) + '%';

        document.getElementById('revenueMediumTotal').textContent = 'R' + formatNumber(totalRevenueMedium, 0);
        document.getElementById('revenueHighTotal').textContent = 'R' + formatNumber(totalRevenueHigh, 0);

        if (!cmaAvgRatePerM2 || cmaAvgRatePerM2 <= 0) {
            displayError('cmaAvgRatePerM2', 'CMA average rate per m² must be a positive number.');
            hasErrors = true;
        }

        if (hasErrors) {
            return; // Stop form submission if validation fails
        }

        if (!currentCosts) {
            alert('Unable to calculate costs for the given purchase price. Please check your input and ensure it falls within the defined ranges.');
            return;
        }

        // Use the deposit amount for analysis
        const depositAmount = document.getElementById('depositAmount').value;

        if (depositAmount) {
            data.deposit = parseFloat(depositAmount);
        } else if (depositPercentage) {
            data.deposit = (parseFloat(depositPercentage) / 100) * purchasePrice;
        } else {
            data.deposit = 0; // Default to 0 if neither is filled
        }

        // Immediately recalculate transfer costs and total capital required
        const { transferCosts, totalCapitalRequired } = recalculateTransferCosts();

        // Validate deposit input
        if (!validateDeposit(purchasePrice)) {
            return; // Stop form submission if validation fails
        }

        // Get deposit value
        data.deposit = getDepositValue(purchasePrice);

        // Calculate deposit percentage
        const depositPercentage = (data.deposit / purchasePrice) * 100;

        // Update UI with initial calculations
        document.getElementById('summaryPurchasePrice').textContent = 'R' + formatNumber(purchasePrice, 0);
        document.getElementById('summaryDeposit').textContent = 'R' + formatNumber(data.deposit, 0);
        document.getElementById('summaryDepositPercentage').textContent = formatNumber(depositPercentage, 2);
        document.getElementById('summaryBondRegistration').textContent = 'R' + formatNumber(bondRegistration, 0);
        document.getElementById('summaryTransferCosts').textContent = 'R' + formatNumber(transferCosts, 0);
        document.getElementById('summaryTotalCapitalRequired').textContent = 'R' + formatNumber(totalCapitalRequired, 0);

        // Fetch analysis results
        fetch('/analysis', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
            },
            body: JSON.stringify(data)
        })
        .then(response => {
            if (!response.ok) {
                throw new Error(`HTTP error! status: ${response.status}`);
            }
            return response.json();
        })
        .then(results => {
            console.log('Analysis results:', results);
            window.analysisResults = results;  // Store the results for later use

            // Handle Infinity values
            for (let metric in results) {
                if (typeof results[metric] === 'object') {
                    for (let year in results[metric]) {
                        if (results[metric][year] === "Infinity") {
                            results[metric][year] = Infinity;
                        }
                    }
                }
            }

            populateAirbnbPerformance(data);
            createAirbnbRevenueChart(data);
            initializeMap(data.address);

            // Populate Deal Summary
            document.getElementById('summaryAddress').textContent = data.address;
            document.getElementById('summaryPropertyDescription').textContent = data.comments || 'No property description provided.';
            document.getElementById('summaryPurchasePrice').textContent = 'R' + formatNumber(purchasePrice, 0);
            document.getElementById('summaryDeposit').textContent = 'R' + formatNumber(data.deposit, 0);
            document.getElementById('summaryDepositPercentage').textContent = formatNumber(depositPercentage, 2);
            document.getElementById('summaryInterestRate').textContent = data.interestRate + '%';
            document.getElementById('summaryFloorArea').textContent = data.floorArea + ' m²';
            document.getElementById('summaryBeds').textContent = data.beds;
            document.getElementById('summaryBaths').textContent = data.baths;
            document.getElementById('summaryParking').textContent = data.parkings;
            document.getElementById('summaryBondRepayment').textContent = 'R' + formatNumber(results.monthlyBondRepayment, 2);
            document.getElementById('summaryBondRegistration').textContent = 'R' + formatNumber(bondRegistration, 0);
            document.getElementById('summaryTransferCosts').textContent = 'R' + formatNumber(transferCosts, 0);
            document.getElementById('summaryTotalCapitalRequired').textContent = 'R' + formatNumber(totalCapitalRequired, 0);
            document.getElementById('summaryNightlyRate').textContent = 'R' + formatNumber(parseFloat(data.nightlyRate), 0);
            document.getElementById('summaryOccupancy').textContent = data.occupancy;

            const ltrGrossYield = results.longTermRental && results.longTermRental[1] ? results.longTermRental[1].grossYield : null;
            const ltrRevenue = results.longTermRental && results.longTermRental[1] ? results.longTermRental[1].revenue : null;

            // Retrieve STR Revenue for Year 1
            const strRevenue = results.revenue && results.revenue[1] ? results.revenue[1] : null;

            // Retrieve LTR Amount from the form data
            const ltrAmount = parseFloat(data.longTermRental);

            // Update the LTR Gross Yield, LTR Revenue, and STR Revenue in the Deal Summary
            if (ltrGrossYield !== null) {
                document.getElementById('summaryLTRGrossYield').textContent = formatNumber(ltrGrossYield, 2) + '%';
            } else {
                document.getElementById('summaryLTRGrossYield').textContent = 'N/A';
            }

            if (ltrRevenue !== null) {
                document.getElementById('summaryLTRRevenue').textContent = 'R' + formatNumber(ltrRevenue, 0);
            } else {
                document.getElementById('summaryLTRRevenue').textContent = 'N/A';
            }

            if (strRevenue !== null) {
                document.getElementById('summarySTRRevenue').textContent = 'R' + formatNumber(strRevenue, 0);
            } else {
                document.getElementById('summarySTRRevenue').textContent = 'N/A';
            }

            if (ltrAmount) {
                document.getElementById('summaryLTRAmount').textContent = 'R' + formatNumber(ltrAmount, 0);
            } else {
                document.getElementById('summaryLTRAmount').textContent = 'N/A';

            const ratePerSqm = purchasePrice / floorArea;
            document.getElementById('summaryRatePerSqm').textContent = 'R' + formatNumber(ratePerSqm, 0) + '/m²';

            // Calculate and display rate difference
            const rateDifferenceElement = document.getElementById('summaryRateDifference');
            const difference = ratePerSqm - cmaAvgRatePerM2;

            // Reset transfer duty option to default
            document.getElementById('transferDutyOption').value = 'with';

            if (!isNaN(difference) && isFinite(difference)) {
                const absDifference = Math.abs(difference);
                const percentageDifference = (difference / cmaAvgRatePerM2) * 100;

                let differenceText = `R${formatNumber(absDifference, 0)}/m² `;
                differenceText += difference < 0 ? 'less' : 'more';
                differenceText += ` (${formatNumber(Math.abs(percentageDifference), 2)}%)`;

                        } else {
                            cell.textContent = formatNumber(value, 2);
                        }
                    } else {
                        console.warn(`Cell not found or data missing for ${metric}${year}`);
                    }
                });
            });

            years.forEach(year => {
                const propertyValue = results.propertyValue[year];
                let previousYearValue;

                if (year === 1) {
                    previousYearValue = purchasePrice;
                } else if (year <= 5) {
                    previousYearValue = results.propertyValue[year - 1];
                } else if (year === 10) {
                    previousYearValue = results.propertyValue[5];
                } else if (year === 20) {
                    previousYearValue = results.propertyValue[10];
                }

                if (propertyValue !== undefined && previousYearValue !== undefined) {
                    const randAppreciation = propertyValue - previousYearValue;
                    const randAppreciationElement = document.getElementById(`randAppreciation${year}`);
                    if (randAppreciationElement) {
                        randAppreciationElement.textContent = 'R' + formatNumber(randAppreciation, 0);
                    } else {
                        console.warn(`Element with id randAppreciation${year} not found`);
                    }
                } else {
                    console.warn(`Missing property value data for year ${year}`);
                }
            });

            // Calculate IRR for each year
            years.forEach(year => {
                const cashFlows = [-results.purchasePrice];  // Initial investment
                for (let i = 1; i <= year; i++) {
                    cashFlows.push(results.annualCashflow[i]);
                }
                // Add property value at the end of the period
                cashFlows[cashFlows.length - 1] += results.propertyValue[year];

                const irr = calculateIRR(cashFlows);
                if (irr !== null) {
                    document.getElementById(`irr${year}`).textContent = formatNumber(irr * 100, 2) + '%';
                } else {
                    document.getElementById(`irr${year}`).textContent = 'N/A';
                }
            });

            // Create charts
            createFinancialChart(results);
            createCashflowChart(results);
            createEquityChart(results);

            // Populate Long-Term Rental Analysis table
            if (results.longTermRental) {
                years.forEach(year => {
                    if (results.longTermRental[year]) {
                        document.getElementById(`ltRevenue${year}`).textContent = 'R' + formatNumber(results.longTermRental[year].revenue, 0);
                        document.getElementById(`ltGrossYield${year}`).textContent = formatNumber(results.longTermRental[year].grossYield, 2) + '%';
                    }
                });
            }

            // Calculate STR Monthly Revenue
            const nightlyRate = parseFloat(data.nightlyRate);
            const occupancy = parseFloat(data.occupancy) / 100;
            const strMonthlyRevenue = nightlyRate * occupancy * 30; // Assuming 30 days in a month

            // Update the STR Monthly Revenue in the Deal Summary
            document.getElementById('summarySTRMonthlyRevenue').textContent = 'R' + formatNumber(strMonthlyRevenue, 0);

            document.getElementById('analysisResults').style.display = 'block';
            // Make sure the Save Property button is visible
            document.getElementById('savePropertyBtn').style.display = 'inline-block';

            // After handling results, recalculate once more to ensure everything is up to date
            const finalCalculation = recalculateTransferCosts();
            document.getElementById('summaryTransferCosts').textContent = 'R' + formatNumber(finalCalculation.transferCosts, 0);
            document.getElementById('summaryTotalCapitalRequired').textContent = 'R' + formatNumber(finalCalculation.totalCapitalRequired, 0);
            })
            .catch(error => {
            console.error('Error:', error);
            alert('An error occurred while analyzing the property. Please check the console for details.');
            });
            });

            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
            },
            body: JSON.stringify({ address, bedrooms })
        });

        console.log('Response status:', response.status);
        console.log('Response headers:', Object.fromEntries(response.headers.entries()));

        const data = await response.json();
        console.log('Received data:', JSON.stringify(data, null, 2));

        if (!response.ok) {
            throw new Error(data.error || `HTTP error! status: ${response.status}`);

        this.textContent = 'Get Revenue Data';

    function addChartToPDF(doc, chartId, y) {
        try {

                const imgWidth = 180; // Adjust as needed
                const imgHeight = (canvas.height * imgWidth) / canvas.width;

                if (y + imgHeight > 280) {

                }

            return y + 10; // Return an incremented y value if there's an error
        }
    }

    function addLogoToPDF(doc, logoUrl, x, y, maxWidth, maxHeight) {
    return new Promise((resolve, reject) => {
        const img = new Image();
        img.onload = function() {
            let width = img.width;
            let height = img.height;
            
            // Calculate aspect ratio
            const aspectRatio = width / height;

            // Adjust dimensions to fit within maxWidth and maxHeight while maintaining aspect ratio
            if (width > maxWidth) {
                width = maxWidth;
                height = width / aspectRatio;
            }
            if (height > maxHeight) {
                height = maxHeight;
                width = height * aspectRatio;
            }

            doc.addImage(this, 'PNG', x, y, width, height);
            resolve(doc);
        };
        img.onerror = reject;
        img.src = logoUrl;
    });
}

function cleanup() {
    // Restore button visibility
    document.getElementById('exportPDFBtn').style.display = 'inline-block';
    document.getElementById('shareBtn').style.display = 'inline-block';

    // Remove modal-related classes from body
    document.body.classList.remove('modal-open');
    document.body.style.overflow = '';
    document.body.style.paddingRight = '';

    // Remove any lingering backdrop
    const backdrop = document.querySelector('.modal-backdrop');
    if (backdrop) {
        backdrop.remove();
    }

    // Close any open modals
    const modals = document.querySelectorAll('.modal');
    modals.forEach(modal => {
        const bsModal = bootstrap.Modal.getInstance(modal);
        if (bsModal) {
            bsModal.hide();
        }
    });

    // Reset any inline styles that might affect scrolling
    document.documentElement.style.overflow = '';
    document.body.style.position = '';
    document.body.style.width = '';
    document.body.style.top = '';

    // Force a window resize event to help recalculate any scroll-related layouts
    window.dispatchEvent(new Event('resize'));
}

        const dt = document.createElement('dt');
        dt.className = 'col-sm-3';
        dt.textContent = detail.label;

}

function updateAIRecommendation(data) {
    document.getElementById('aiRecommendationText').textContent = data.ai_recommendation;
    
    const percentilesList = document.getElementById('percentilesList');
    percentilesList.innerHTML = '';
    
    for (const [key, value] of Object.entries(data.percentiles)) {
        if (key !== 'occupancy') {
            const li = document.createElement('li');
            li.textContent = `${key}: R${value}`;
            percentilesList.appendChild(li);
        }
    }

    document.getElementById('aiRecommendation').style.display = 'block';
}
