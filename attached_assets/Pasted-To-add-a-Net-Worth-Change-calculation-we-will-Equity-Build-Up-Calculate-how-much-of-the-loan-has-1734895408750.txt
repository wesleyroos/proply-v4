To add a Net Worth Change calculation, we will:

Equity Build-Up:
Calculate how much of the loan has been paid down over time.
Cumulative Cash Flow:
Sum the cash flow from the property over the years.
Property Appreciation:
Factor in the annual appreciation of the property's value.
Net Worth Change:
Combine the above components to calculate the user's net worth change over time.

// Property Appreciation Calculation
const annualAppreciationRate = 6 / 100; // Assuming 6% annual appreciation
const propertyAppreciation = {
  year1: data.purchasePrice * Math.pow(1 + annualAppreciationRate, 1) - data.purchasePrice,
  year2: data.purchasePrice * Math.pow(1 + annualAppreciationRate, 2) - data.purchasePrice,
  year3: data.purchasePrice * Math.pow(1 + annualAppreciationRate, 3) - data.purchasePrice,
  year4: data.purchasePrice * Math.pow(1 + annualAppreciationRate, 4) - data.purchasePrice,
  year5: data.purchasePrice * Math.pow(1 + annualAppreciationRate, 5) - data.purchasePrice,
  year10: data.purchasePrice * Math.pow(1 + annualAppreciationRate, 10) - data.purchasePrice,
  year20: data.purchasePrice * Math.pow(1 + annualAppreciationRate, 20) - data.purchasePrice,
};

// Loan Amortization / Equity Build-Up Calculation
const equityBuildUp = {
  year1: (loanAmount - (monthlyBondRepayment * 12) * 0.25), // Approximate principal repayment as 25% of repayment
  year2: (loanAmount - (monthlyBondRepayment * 24) * 0.25),
  year3: (loanAmount - (monthlyBondRepayment * 36) * 0.25),
  year4: (loanAmount - (monthlyBondRepayment * 48) * 0.25),
  year5: (loanAmount - (monthlyBondRepayment * 60) * 0.25),
  year10: (loanAmount - (monthlyBondRepayment * 120) * 0.25),
  year20: (loanAmount - (monthlyBondRepayment * 240) * 0.25),
};

// Cumulative Cash Flow Calculation
const cumulativeCashFlow = {
  year1: netOperatingIncome?.year1 || 0,
  year2: (netOperatingIncome?.year1 || 0) + (netOperatingIncome?.year2 || 0),
  year3: (netOperatingIncome?.year1 || 0) + (netOperatingIncome?.year2 || 0) + (netOperatingIncome?.year3 || 0),
  year4: (netOperatingIncome?.year1 || 0) + (netOperatingIncome?.year2 || 0) + (netOperatingIncome?.year3 || 0) + (netOperatingIncome?.year4 || 0),
  year5: (netOperatingIncome?.year1 || 0) + (netOperatingIncome?.year2 || 0) + (netOperatingIncome?.year3 || 0) + (netOperatingIncome?.year4 || 0) + (netOperatingIncome?.year5 || 0),
  year10: Array.from({ length: 10 }, (_, i) => netOperatingIncome?.[`year${i + 1}`] || 0).reduce((a, b) => a + b, 0),
  year20: Array.from({ length: 20 }, (_, i) => netOperatingIncome?.[`year${i + 1}`] || 0).reduce((a, b) => a + b, 0),
};

// Net Worth Change Calculation
const netWorthChange = {
  year1: equityBuildUp.year1 + cumulativeCashFlow.year1 + propertyAppreciation.year1,
  year2: equityBuildUp.year2 + cumulativeCashFlow.year2 + propertyAppreciation.year2,
  year3: equityBuildUp.year3 + cumulativeCashFlow.year3 + propertyAppreciation.year3,
  year4: equityBuildUp.year4 + cumulativeCashFlow.year4 + propertyAppreciation.year4,
  year5: equityBuildUp.year5 + cumulativeCashFlow.year5 + propertyAppreciation.year5,
  year10: equityBuildUp.year10 + cumulativeCashFlow.year10 + propertyAppreciation.year10,
  year20: equityBuildUp.year20 + cumulativeCashFlow.year20 + propertyAppreciation.year20,
};
