Here is the correct way to dynamically update the marker with google.maps.marker.AdvancedMarkerElement:

useEffect(() => {
  const updateMarker = async () => {
    if (!map || !window.google?.maps) {
      console.log('Map not ready yet');
      return;
    }

    // If no address, reset to default location
    if (!address?.trim()) {
      const defaultLocation = { lat: -33.918861, lng: 18.423300 }; // Cape Town
      map.setCenter(defaultLocation);
      map.setZoom(13);
      if (marker) marker.map = null; // Hide marker
      return;
    }

    try {
      console.log('Geocoding address:', address);
      const geocoder = new window.google.maps.Geocoder();

      const result = await new Promise((resolve, reject) => {
        geocoder.geocode({ address }, (results, status) => {
          if (status === 'OK' && results?.[0]) {
            resolve(results[0]);
          } else {
            reject(new Error(`Geocoding failed: ${status}`));
          }
        });
      });

      const location = result.geometry.location;
      console.log('Found location:', location.toString());

      map.setCenter(location);
      map.setZoom(16);

      // Create or update the AdvancedMarkerElement
      if (marker) {
        marker.position = location;
        marker.map = map;
      } else {
        const newMarker = new window.google.maps.marker.AdvancedMarkerElement({
          map: map,
          position: location,
          title: 'Property Location',
        });
        setMarker(newMarker);
      }

      console.log('Marker updated successfully at:', location.toString());
    } catch (error) {
      console.error('Error updating marker:', error);
      if (marker) marker.map = null;
    }
  };

  updateMarker();
}, [address, map]);
