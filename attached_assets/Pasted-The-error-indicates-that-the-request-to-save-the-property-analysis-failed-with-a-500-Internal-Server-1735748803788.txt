import { db } from '@/lib/db';

export default async function handler(req, res) {
  if (req.method !== 'POST') {
    return res.status(405).json({ error: 'Method not allowed' });
  }

  try {
    const { userId, address, purchasePrice, ...rest } = req.body;

    if (!userId || !address || !purchasePrice) {
      return res.status(400).json({ error: 'Missing required fields' });
    }

    const result = await db.propertyAnalysis.create({
      data: { userId, address, purchasePrice, ...rest },
    });

    return res.status(200).json(result);
  } catch (error) {
    console.error('Error saving property analysis:', error);
    return res.status(500).json({ error: 'Failed to save property analysis' });
  }
}
