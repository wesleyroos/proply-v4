I can help you add PDF export functionality using the react-pdf library or jsPDF. Here's how we can implement this:

typescript

Copy
import { Button } from "@/components/ui/button";
import { Download } from "lucide-react";
import jsPDF from "jspdf";
import "jspdf-autotable"; // For table support

// Add to your existing imports

function PropertyComparisonModal({ ... }) {
  // Add this function to generate and download PDF
  const generatePDF = () => {
    const doc = new jsPDF();
    
    // Add Proply logo
    doc.addImage("/proply-logo-auth.png", "PNG", 10, 10, 50, 20);
    
    // Add title
    doc.setFontSize(20);
    doc.text("Property Comparison Report", 10, 40);
    
    // Add date
    doc.setFontSize(12);
    doc.text(`Generated on: ${new Date().toLocaleDateString()}`, 10, 50);
    
    // Properties Overview
    doc.setFontSize(14);
    doc.text("Properties Overview", 10, 70);
    
    // Create overview table
    const overviewData = displayProperties.map(p => [
      p.address,
      formatter.format(p.purchasePrice),
      `${p.floorArea} m²`,
      `${p.shortTermGrossYield}%`,
      `${p.longTermGrossYield}%`
    ]);
    
    doc.autoTable({
      head: [["Address", "Price", "Size", "ST Yield", "LT Yield"]],
      body: overviewData,
      startY: 80,
      styles: { fontSize: 10 },
      headStyles: { fillColor: [27, 163, 255] } // Proply blue
    });
    
    // Add metrics tables by category
    Object.entries(categories).forEach(([category, { title }], index) => {
      const startY = doc.lastAutoTable.finalY + 20;
      
      doc.setFontSize(14);
      doc.text(title, 10, startY);
      
      const categoryMetrics = metrics.filter(m => m.category === category);
      const metricData = categoryMetrics.map(metric => {
        return [
          metric.label,
          ...displayProperties.map(prop => metric.format(metric.getValue(prop)))
        ];
      });
      
      doc.autoTable({
        head: [["Metric", ...displayProperties.map(p => p.address)]],
        body: metricData,
        startY: startY + 10,
        styles: { fontSize: 10 },
        headStyles: { fillColor: [27, 163, 255] }
      });
    });
    
    // Add performance chart
    // Note: You'll need to convert your chart to an image first
    // One way is to use html2canvas on your chart component
    
    doc.save("property-comparison-report.pdf");
  };

  return (
    <Dialog open={open} onOpenChange={onOpenChange}>
      <DialogContent className="max-w-6xl max-h-[90vh] bg-white dark:bg-gray-900">
        <DialogHeader className="px-6 py-4 border-b border-gray-200 dark:border-gray-800">
          <div className="flex justify-between items-center">
            <DialogTitle className="text-xl font-semibold text-gray-900 dark:text-white">
              Property Comparison
            </DialogTitle>
            <div className="flex items-center gap-4">
              <Button
                onClick={generatePDF}
                variant="outline"
                className="flex items-center gap-2"
              >
                <Download className="h-4 w-4" />
                Export Report
              </Button>
              <img
                src="/proply-logo-auth.png"
                alt="Proply Logo"
                className="h-8 object-contain"
              />
            </div>
          </div>
        </DialogHeader>
        {/* Rest of your modal content... */}
      </DialogContent>
    </Dialog>
  );
}
To make this work well, you'll need to:

Install required dependencies:
bash

Copy
npm install jspdf jspdf-autotable html2canvas
For including the chart in the PDF, add this function:
typescript

Copy
const exportChart = async () => {
  const chartElement = document.querySelector('.recharts-wrapper');
  if (!chartElement) return null;
  
  const canvas = await html2canvas(chartElement);
  return canvas.toDataURL('image/png');
};
Then in your PDF generation, add:
typescript

Copy
// Inside generatePDF function
const chartImage = await exportChart();
if (chartImage) {
  doc.addPage();
  doc.text("Performance Overview", 10, 20);
  doc.addImage(chartImage, 'PNG', 10, 30, 190, 100);
}
